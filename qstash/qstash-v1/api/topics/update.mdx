---
title: "Update Topic"
description: "This endpoint updates the existing topic with the specified ID."
api: "PUT https://qstash.upstash.io/v1/topics/:topicId"
authMethod: "bearer"
---

## Request

<ParamField path="topicId" required>
  The id or name of the topic
</ParamField>

<ParamField body="name" type="string" required>
  The new name of the topic Must only contain alphanumeric characters,
  underscores, dashes and periods. regex: ^[a-zA-Z0-9-_.]+$
</ParamField>

## Response

<ResponseField name="endpoints" type="Array<Endpoint>" required>
    <Expandable title="Endpoint">
        <ResponseField name="endpointId" type="string" required >
            The id of the endpoint.
        </ResponseField>

        <ResponseField name="topicId" type="string" required >
            The id of the topic this endpoint subscribes to.
        </ResponseField>

        <ResponseField name="url" type="string" required>
            The url where all requests will be sent to.
        </ResponseField>
    </Expandable>

</ResponseField>

<ResponseField name="name" type="string" required>
  A user given name must only contain alphanumeric, hyphen, underscore and
  periods.
</ResponseField>

<ResponseField name="topicId" type="string" required>
  Id for this topic
</ResponseField>

<RequestExample>

```shell curl
curl -X PUT https://qstash.upstash.io/v1/topics/:topicId\
  -H "Authorization: Bearer <token>" \
  -d '{ "name": "test2" }'
```

</RequestExample>
